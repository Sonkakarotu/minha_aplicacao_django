# IMPORTANDO A CLASSE CHOICE, QUESTION:
>>> from polls.models import Choice, Question

# VER TODAS AS PERGUNDAS:
>>> Question.objects.all()

# IMPORTAÇÃO DO TIMEZONE:
>>> from django.utils import timezone

# DECLARANDO UMA VARIAVEL PARA RECEBER A PERGUNTA:
>>> q = Question(question_text="O que ha de novo?", pub_date=timezone.now())

# SALVANDO ESSA VARIAVEL NO BANCO:
>>> q.save()

# AGORA VAMOS VER A ID DESSA VARIAVEL NO BANCO:
>>> q.id

# EXIBINDO NOSSO TEXTO:
>>> q.question_text

# EXIBINDO A HORA:
>>> q.pub_date

# ALTERANDO NOSSO TEXTO:
>>> q.question_text = "O que para cima?"

# SALVANDO ESSA VARIAVEL NO BANCO:
>>> q.save()

# VER TODAS AS PERGUNDAS:
>>> Question.objects.all()

# MOSTRANDO POR FILTROS DE ID:
>>> Question.objects.filter(id=1)

# MOSTRANDO POR FILTROS DE TEXTO:
>>> Question.objects.filter(question_text__startswith='O que')

# PARA VER AS PERGUNTAS POR ANO PRIMEIRO IMPORTE TIMEZONE:
>>> from django.utils import timezone

# DECLARAÇÃO DA VARIAVEL PARA RECEBER ANO:
>>> current_year = timezone.now().year

# PEGANDO A DATA EO ANO PARA EXIBIR TEXTO:
>>> Question.objects.get(pub_date__year=current_year)
